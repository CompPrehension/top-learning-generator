# from SQLite
CREATE TABLE questions (
    id                             INTEGER PRIMARY KEY ASC AUTOINCREMENT
                                           NOT NULL
                                           DEFAULT (0),
    name                           TEXT,
    template_id                    INTEGER REFERENCES templates (id)
                                           NOT NULL,
    q_graph                        TEXT,
    q_data_graph                   TEXT,
    tag_bits                       INTEGER,
    concept_bits                   INTEGER,
    law_bits                       INTEGER,
    violation_bits                 INTEGER,
    solution_structural_complexity REAL,
    integral_complexity            REAL,
    solution_steps                 INTEGER,
    distinct_errors_count          INTEGER,
    _stage                         INTEGER,
    _version                       INTEGER
);

# to MySQL
CREATE TABLE questions_meta (
    id                             INTEGER  NOT NULL AUTO_INCREMENT,
    name                           TEXT,
    domain_shortname               VARCHAR(45) NOT NULL DEFAULT "ctrl_flow",  # added
    template_id                    INTEGER NOT NULL,
    q_graph                        TEXT,
    q_data_graph                   TEXT,
    tag_bits                       INTEGER,
    concept_bits                   INTEGER,
    law_bits                       INTEGER,
    violation_bits                 INTEGER,
    solution_structural_complexity REAL,
    integral_complexity            REAL,
    solution_steps                 INTEGER,
    distinct_errors_count          INTEGER,
    _stage                         INTEGER,
    _version                       INTEGER,
  PRIMARY KEY (`id`)
);
